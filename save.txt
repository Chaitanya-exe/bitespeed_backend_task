model Contact {
  id             Int       @id @default(autoincrement())
  email          String?
  phoneNumber    String?
  linkedId       Int?
  linkPrecedence LinkPrecedence
  createdAt      DateTime  @default(now())
  updatedAt      DateTime  @updatedAt
  deletedAt      DateTime?

  @@index([email])
  @@index([phoneNumber])
  @@index([linkedId])
}

enum LinkPrecedence {
  primary
  secondary
}
